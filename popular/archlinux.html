Arch Linux（KDE桌面） 快速上手

常见报错

# pacman 数据库被锁定
# ::Synchronizing package databases.. 
# error: failed to synchronize all databases(unable to lock database)
# 如果自己没有运行的程序，删除锁定文件
sudo rm -f /var/lib/pacman/db.lck

# 错误:无法删除 /var/cache/pacman/pkg/download-xxxxx
# 使用 rm 删除
sudo rm -rf /var/cache/pacman/pkg/download-*

设置中文（KDE 桌面）

  • 保留系统 en_US
  • 显示语言设置为中文

编辑 /etc/locale.gen 文件，取消 zh_CN 的注释

#zh_CN.GB18030 GB18030
#zh_CN.GBK GBK
#zh_CN.UTF-8 UTF-8
#zh_CN GB312

生成配置

sudo locale-gen

设置显示语言

  • 打开设置 System Settings
  • Region & Language
  • 选择 Language 选项并点击 Modify
  • 点击 Change Language
  • 拉到最后选择简体中文并点击 Apply
  • 重启

# 依赖安装:
# sudo pacman -S eza bat btop zoxide starship fzf ripgrep fd duf dust tldr ouch zsh-syntax-highlighting zsh-autosuggestions pacman-contrib reflector

if [[ -n ${ZDOTDIR:-$HOME}/.zcompdump(#qN.mh+24) ]]; then
  compinit
else
  compinit -C

setopt HIST_IGNORE_ALL_DUPS  # 忽略重复命令
setopt HIST_IGNORE_SPACE     # 忽略以空格开头的命令

# 补全菜单样式美化
zstyle ':completion:*' menu select
zstyle ':completion:*' matcher-list 'm:{a-zA-Z}={A-Za-z}' # 忽略大小写
zstyle ':completion:*:descriptions' format '%F{green}-- %d --%f'
zstyle ':completion:*:targets' list-colors '=(#b)(*)/=34'

# 加载插件
[[ -f /usr/share/zsh/plugins/zsh-syntax-highlighting/zsh-syntax-highlighting.zsh ]] && source /usr/share/zsh/plugins/zsh-syntax-highlighting/zsh-syntax-highlighting.zsh

# FZF 模糊搜索集成
if [[ -f /usr/share/fzf/key-bindings.zsh ]]; then
    source /usr/share/fzf/key-bindings.zsh
    source /usr/share/fzf/completion.zsh

    # 使用 fd 代替 find 作为 FZF 的后端，自动忽略 .git 和 node_modules

    export FZF_DEFAULT_OPTS="--height 40% --layout=reverse --border --inline-info"
    export FZF_CTRL_T_OPTS="--preview 'bat --color=always --line-range :500 {}'"
    export FZF_ALT_C_OPTS="--preview 'eza --tree --icons {} | head -200'"
fi

# 万能解压
ex() {
    if command -v ouch > /dev/null; then
        ouch decompress "$@"
    else
        if [ -f "$1" ] ; then
            case $1 in
                *.tar.bz2)   tar xvf "$1"     ;;
                *.tar.gz)    tar xvf "$1"     ;;
                *.bz2)       bunzip2 "$1"     ;;
                *.rar)       unrar x "$1"     ;;
                *.gz)        gunzip "$1"      ;;
                *.tar)       tar xvf "$1"     ;;
                *.tbz2)      tar xvf "$1"     ;;
                *.tgz)       tar xvf "$1"     ;;
                *.zip)       unzip "$1"       ;;
                *.7z)        7z x "$1"        ;;
                *)           echo "'$1' cannot be extracted via ex()" ;;
            esac
        else
            echo "'$1' is not a valid file"
        fi
    fi
}

# 快速创建目录并进入
mkcd() {
    mkdir -p "$1" && cd "$1"
}

编辑 ~/.config/fish/config.fish 文件

# ~/.config/fish/config.fish
# -------------------------------
# 基础环境变量
set -gx EDITOR nvim
set -gx TERMINAL alacritty
set -gx BAT_THEME TwoDark
set -gx PATH $HOME/.local/bin $PATH

# -------------------------------
# starship prompt
if type -q starship
    eval (starship init fish)
end

# -------------------------------
# zoxide: 智能跳目录
if type -q zoxide
    eval (zoxide init fish)
end

# -------------------------------
# direnv: 自动加载目录环境变量
if type -q direnv
    eval (direnv hook fish)
end

# -------------------------------
# atuin: 智能历史
if type -q atuin
    set -gx ATUIN_NOBIND true
    atuin init fish | source
    bind \cr _atuin_search
end

# -------------------------------
# fzf 全局设置
set -gx FZF_DEFAULT_COMMAND 'fd --type f --hidden --follow --exclude .git'
set -gx FZF_DEFAULT_OPTS '--height 40% --reverse --border --ansi'

# -------------------------------
# 常用 alias
alias ls='eza --icons'
alias ll='eza -lah --icons --git'
alias la='eza -a --icons'
alias cat='bat'
alias grep='rg'
alias v='nvim'
alias g='git'

# -------------------------------
# 函数：fzf 打开文件
function fzf_open -d "用 fzf 查找文件并用 $EDITOR 打开"
    set -l file (fzf --preview 'bat --style=numbers --color=always {}')
    if test -n "$file"
        $EDITOR "$file"
    end
end
bind \cf 'fzf_open'

# -------------------------------
# 函数：fzf + zoxide 跳目录
function fzf_z -d "用 zoxide + fzf 跳转目录"
    set -l sel (zoxide query -ls | fzf --height 40% --reverse --ansi)
    if test -n "$sel"
        set -l target (string match -r '[^\t]+$' -- $sel)
        cd "$target"
    end
end
bind \cg 'fzf_z'

# -------------------------------
# 函数：全局内容搜索（rg + fzf）
function fzf_rg -d "ripgrep + fzf 全局搜索"
    set -l query (fzf --phony --query "$argv" \
        --bind "change:reload:rg --line-number --no-heading --color=always {q} || true" \
        --preview 'bat --style=numbers --color=always {1} --highlight-line {2}' \
        --delimiter : --nth 3..)
    if test -n "$query"
        set -l file (echo $query | cut -d: -f1)
        set -l line (echo $query | cut -d: -f2)
        $EDITOR +$line "$file"
    end
end
bind \cs 'fzf_rg'

# -------------------------------
# 函数：broot 快速跳转
function br -d "启动 broot"
    broot
end

# -------------------------------
# Git Delta 作为 pager
set -gx GIT_PAGER 'delta'

# -------------------------------
# End of config.fish

登录卡死

MD 显卡在 KDE Wayland 下登录卡死，通常是
由于 电源管理冲突、Mesa 驱动回归或 DDC/CI（显示器控制协议）读取错误导致的。
可以尝试以下三个具体的修复方案：

方案 1：

禁用 PowerDevil 的 DDC/CI 检测
这是一个非常常见的 AMD/Wayland 冲突点。某些显示器在读取亮度信息时会导致整个 Plasma 进程挂起。
按 Ctrl+Alt+F3 进入终端并登录。

输入命令：
systemctl --user edit plasma-powerdevil.service

在打开的编辑器中添加以下两行：

[Service]
Environment="POWERDEVIL_NO_DDCUTIL=1"

然后重启：sudo reboot。

方案 2：

添加内核引导参数（针对 AMD 节能机制）
部分 AMD 显卡在 Wayland 下切换电源状态时会触发 GPU 超时。可以通过修改引导参数强制 显卡保持活跃：

编辑 GRUB 配置文件： sudo nano /etc/default/grub
在 GRUB_CMDLINE_LINUX_DEFAULT 这一行的引号内加入：
amdgpu.runpm=0 amdgpu.dcdebugmask=0x10
